#include <stdio.h>
#include <ctype.h>   
int main() {
    char str[100];
    int i;
    printf("Enter a string: ");
    fgets(str, sizeof(str), stdin);  
    printf("\nString in Uppercase: ");
    for (i = 0; str[i] != '\0'; i++) {
        putchar(toupper(str[i]));
    }
    printf("\nString in Lowercase: ");
    for (i = 0; str[i] != '\0'; i++) {
        putchar(tolower(str[i]));
    }
    return 0;
}

output:
Enter a string: Vijay

String in Uppercase: VIJAY

String in Lowercase: vijay
